# Contador de frecuencia de palabras
from collections import Counter
import re

def contar_palabras(texto, top_n=10):
    """
    Cuenta la frecuencia de palabras en un texto.
    
    Args:
        texto (str): Texto a analizar
        top_n (int): N칰mero de palabras m치s frecuentes a retornar
    
    Returns:
        Counter: Frecuencia de palabras
    """
    # Convertir a min칰sculas y eliminar caracteres especiales
    texto_limpio = re.sub(r'[^\w\s]', '', texto.lower())
    palabras = texto_limpio.split()
    
    # Contar frecuencia
    frecuencia = Counter(palabras)
    
    return frecuencia.most_common(top_n)

# Ejemplo de uso
